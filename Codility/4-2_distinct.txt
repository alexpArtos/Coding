class Solution {
    public int solution(int[] A) {
        // write your code in C# 5.0 with .NET 4.5 (Mono)
        int n = A.Length;
        if (n == 0) return 0;
        Array.Sort(A);
        int count = 0;
        for (int i = 1; i < n; i++) {
            if (A[i] > A[i-1]) count++;   
        }
        return count+1;
    }
}
